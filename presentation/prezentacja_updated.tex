\section{Introduction}
\begin{frame}
\frametitle{Problem and Goal}
\textbf{Goal:} compress image classifiers while preserving as much accuracy as possible.
\vspace{0.25cm}

\textbf{Project question:}
\begin{itemize}
    \item Can a small student model learn better representations when guided by a larger teacher?
    \item When does embedding distillation help more than plain supervised training?
\end{itemize}
\vspace{0.25cm}

\textbf{Practical motivation:}
\begin{itemize}
    \item teacher = strong but expensive
    \item student = cheaper to deploy
    \item distillation = training-time transfer of teacher knowledge into student weights
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Knowledge Distillation as a Concept}
\textbf{Classical distillation:}
\begin{enumerate}
    \item Train a large \textbf{teacher}.
    \item Freeze the teacher.
    \item Train a smaller \textbf{student} using true labels and additional teacher guidance.
\end{enumerate}
\vspace{0.25cm}

\textbf{Key idea:}
\begin{itemize}
    \item labels only say which class is correct
    \item teacher outputs/features also encode similarity structure, uncertainty, and representation geometry
\end{itemize}
\vspace{0.25cm}

\textbf{Important:} at deployment time we use \textbf{only the student}. The teacher is a training aid, not a runtime dependency.
\end{frame}

\begin{frame}
\frametitle{Why Embedding Distillation?}
\textbf{Logit distillation:}
\begin{itemize}
    \item match teacher class probabilities
\end{itemize}
\vspace{0.2cm}

\textbf{Embedding distillation:}
\begin{itemize}
    \item match internal feature vectors before the classifier
    \item encourage student to organize images in a teacher-like latent space
\end{itemize}
\vspace{0.2cm}

\textbf{Hypothesis:}
\begin{itemize}
    \item stronger feature supervision should help weaker students more than stronger students
    \item effect should become clearer on harder datasets and with larger teacher-student capacity gap
\end{itemize}
\end{frame}

\section{Project Design}
\begin{frame}
\frametitle{Datasets and Experimental Progression}
\textbf{Supported datasets in the codebase:}
\begin{itemize}
    \item CIFAR-10
    \item CIFAR-100
    \item Tiny ImageNet
\end{itemize}
\vspace{0.25cm}

\textbf{How they are used:}
\begin{itemize}
    \item CIFAR-10: smoke tests and initial debugging
    \item CIFAR-100: optional harder intermediate benchmark
    \item Tiny ImageNet: main final benchmark, chosen because the task is harder and the teacher-student gap is more meaningful
\end{itemize}
\vspace{0.25cm}

\textbf{Tiny ImageNet setup used in Colab:}
\begin{itemize}
    \item 200 classes, 64x64 images
    \item train split further divided into 90k train / 10k validation
    \item official validation set used as held-out test set
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Data Pipeline and Augmentations}
\textbf{Preprocessing pipeline:}
\begin{itemize}
    \item optional resize to target image size
    \item convert to tensor
    \item dataset-specific normalization
\end{itemize}
\vspace{0.2cm}

\textbf{Normalization:}
\begin{itemize}
    \item CIFAR-10/100 mean/std: $(0.4914, 0.4822, 0.4465)$ / $(0.2023, 0.1994, 0.2010)$
    \item Tiny ImageNet uses ImageNet normalization: $(0.485, 0.456, 0.406)$ / $(0.229, 0.224, 0.225)$
\end{itemize}
\vspace{0.2cm}

\textbf{Augmentation presets from the code:}
\begin{itemize}
    \item \texttt{none}: resize + normalize only
    \item \texttt{basic}: \texttt{RandomCrop(image\_size, padding=max(4, image\_size/8))} + horizontal flip
    \item \texttt{strong}: basic + \texttt{RandAugment(num\_ops=2, magnitude=9)} + \texttt{RandomErasing(p=0.25)}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Model Architecture}
\textbf{Unified abstraction used in the repository:}
\[
\text{backbone} \rightarrow \text{global pooling} \rightarrow \text{embedding} \rightarrow \text{classifier}
\]
\vspace{0.2cm}

\textbf{Teacher:}
\begin{itemize}
    \item ResNet-50
    \item embedding dimension: 2048
    \item initialized from ImageNet weights, then fine-tuned on the target dataset
\end{itemize}
\vspace{0.2cm}

\textbf{Students:}
\begin{itemize}
    \item ResNet-18, embedding dim 512
    \item MobileNetV3-small, embedding dim 576
    \item Tiny CNN, embedding dim 128
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{How Embeddings Are Wired into Training}
\textbf{Per training batch:}
\begin{enumerate}
    \item Same image goes through teacher and student.
    \item Teacher returns logits and embedding.
    \item Student returns logits and embedding.
    \item Student embedding is projected into teacher embedding space.
    \item Loss is computed from labels and/or teacher targets.
\end{enumerate}
\vspace{0.2cm}

\textbf{Projector used when dimensions differ:}
\begin{itemize}
    \item if dims match: identity
    \item otherwise: \texttt{Linear(student\_dim, hidden)} $\rightarrow$ ReLU $\rightarrow$ \texttt{Linear(hidden, teacher\_dim)}
    \item hidden dimension: $\max(2d_s,\ d_t)$ where $d_s$ is student embedding size and $d_t$ is teacher embedding size
\end{itemize}
\vspace{0.2cm}

\textbf{Inference:}
\begin{itemize}
    \item only the student classifier is needed for prediction
    \item teacher embeddings are not required during deployment
\end{itemize}
\end{frame}

\section{Training Objective}
\begin{frame}
\frametitle{Loss Functions}
\textbf{Classification loss:}
\[
L_{cls} = \text{CrossEntropy}(z_s, y)
\]

\textbf{Embedding loss:}
\begin{itemize}
    \item MSE loss
    \item cosine embedding loss
    \item optional L2 normalization of teacher and projected student embeddings before the loss
\end{itemize}

\textbf{Logit distillation loss:}
\[
L_{logit} = KL\left(\log \sigma(z_s/T), \sigma(z_t/T)\right)\cdot T^2
\]
\vspace{0.15cm}

\textbf{Combined objective:}
\[
L = w_{cls}L_{cls} + w_{emb}L_{emb} + w_{logit}L_{logit}
\]
\end{frame}

\begin{frame}
\frametitle{Training Configuration}
\textbf{Implementation defaults in the code:}
\begin{itemize}
    \item optimizer: AdamW
    \item learning rate: \texttt{3e-4}
    \item weight decay: \texttt{1e-4}
    \item scheduler: cosine annealing over epochs
    \item mixed precision on CUDA, TF32 enabled, optional \texttt{channels\_last}
\end{itemize}
\vspace{0.2cm}

\textbf{CLI defaults:}
\begin{itemize}
    \item teacher: 5 epochs, batch size 64, image size 160
    \item baseline/distillation: 3 epochs, batch size 64, image size 160
    \item default augmentation: \texttt{basic}
\end{itemize}
\vspace{0.2cm}

\textbf{Actual Colab experiment presets:}
\begin{itemize}
    \item CIFAR notebook: image size 64, batch size 96, 2 teacher epochs, 3 student epochs
    \item Tiny ImageNet notebook: image size 64, batch size 128, 6 teacher epochs, 8 student epochs
\end{itemize}
\end{frame}

\section{Experiments}
\begin{frame}
\frametitle{Core Experimental Matrix}
\textbf{Baseline comparisons:}
\begin{itemize}
    \item train each student with classification loss only
    \item compare against distilled counterpart under the same data pipeline
\end{itemize}
\vspace{0.2cm}

\textbf{Students considered:}
\begin{itemize}
    \item ResNet-18
    \item MobileNetV3-small
    \item Tiny CNN
\end{itemize}
\vspace{0.2cm}

\textbf{Why Tiny CNN matters:}
\begin{itemize}
    \item it creates a much larger capacity gap relative to ResNet-50
    \item this is where embedding distillation is most likely to show clear gains
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Expected Interpretation}
\textbf{Questions we want the final results to answer:}
\begin{itemize}
    \item Does embedding distillation outperform baseline training?
    \item Is embedding supervision alone enough, or does it need classification loss?
    \item Does logit distillation help on top of embeddings?
    \item Does normalization improve alignment and stability?
    \item Which student profits most from teacher guidance?
\end{itemize}
\vspace{0.2cm}

\textbf{Working expectation:}
\begin{itemize}
    \item gains should be small on easy datasets or strong students
    \item gains should be clearer on Tiny ImageNet and for Tiny CNN / MobileNetV3-small
\end{itemize}
\end{frame}

\section{Summary}
\begin{frame}
\frametitle{Summary}
\textbf{What is implemented:}
\begin{itemize}
    \item full teacher-baseline-distillation pipeline
    \item multiple datasets and student architectures
    \item embedding and logit distillation objectives
    \item ablations, metrics, visualizations, and presentation-ready artifacts
\end{itemize}
\vspace{0.2cm}

\textbf{Main technical takeaway:}
\begin{itemize}
    \item embedding distillation is a training-time constraint on student representations
    \item the deployed model remains lightweight because only the student is used at inference
\end{itemize}
\vspace{0.2cm}

\textbf{Final claim to validate experimentally:}
\begin{itemize}
    \item distillation is most valuable when the dataset is hard enough and the student is small enough
\end{itemize}
\end{frame}
