\begin{frame}
\frametitle{Cel projektu}
\begin{itemize}
    \item Celem projektu było sprawdzenie, czy mniejszy model może poprawić wyniki dzięki destylacji wiedzy z większego modelu.
    \item Skupiliśmy się na klasyfikacji obrazów oraz porównaniu kilku wariantów uczenia studenta.
    \item Ze względu na ograniczenia czasowe i obliczeniowe finalne eksperymenty wykonano na zbiorze \textbf{CIFAR-100}, a nie na Tiny ImageNet.
\end{itemize}
\vspace{0.2cm}
\textbf{Pytanie badawcze:} czy embedding distillation rzeczywiście pomaga, czy lepszym kierunkiem jest klasyczna destylacja logitów?
\end{frame}

\begin{frame}
\frametitle{Dane i konfiguracja}
\textbf{Zbiór danych:}
\begin{itemize}
    \item CIFAR-100
    \item 100 klas
    \item obrazy o rozmiarze 64x64 po przeskalowaniu
\end{itemize}
\vspace{0.2cm}

\textbf{Ustawienia treningu:}
\begin{itemize}
    \item augmentacja: \texttt{basic} = \texttt{RandomCrop + HorizontalFlip}
    \item batch size: 96
    \item optymalizator: AdamW
    \item learning rate: \texttt{3e-4}
    \item teacher: 15 epok
    \item student: 20 epok
\end{itemize}
\vspace{0.2cm}

\textbf{Metryki:} Accuracy oraz Macro-F1.
\end{frame}

\begin{frame}
\frametitle{Modele i warianty}
\textbf{Teacher:}
\begin{itemize}
    \item ResNet-50
    \item inicjalizacja wagami ImageNet
    \item najlepszy wynik walidacyjny: \textbf{81.82\%}
\end{itemize}
\vspace{0.2cm}

\textbf{Studenci:}
\begin{itemize}
    \item ResNet-18
    \item tiny\_cnn
\end{itemize}
\vspace{0.2cm}

\textbf{Porównywane warianty:}
\begin{itemize}
    \item baseline: tylko Cross-Entropy
    \item embed+cls: strata klasyfikacyjna + dopasowanie embeddingow
    \item embed only
    \item logit+cls
    \item embed+logit+cls
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Jak wyglądała destylacja}
\textbf{Embedding distillation:}
\begin{itemize}
    \item student i teacher dostają ten sam obraz
    \item teacher zwraca logity oraz embedding
    \item student zwraca logity oraz embedding
    \item embedding studenta jest porównywany z embeddingiem teacher'a
\end{itemize}
\vspace{0.2cm}

\textbf{W finalnych wynikach:}
\begin{itemize}
    \item nie użyto PCA bottleneck w końcowym notebooku (\texttt{DISTILL\_DIM = 0})
    \item dla embedding KD zastosowano MSE + normalizację embeddingów
    \item dla logit KD użyto temperatury \texttt{T = 4}
\end{itemize}
\vspace{0.2cm}

\textbf{Wniosek praktyczny:} sprawdziliśmy nie tylko sam embedding KD, ale też czy lepiej działa destylacja logitów.
\end{frame}

\vspace{0.25cm}

\begin{itemize}
    \item Dla \textbf{ResNet-18} embedding distillation dała tylko niewielką poprawę: \textbf{+0.76 p.p.}
    \item Dla \textbf{tiny\_cnn} nie zaobserwowano poprawy.
    \item Sama hipoteza, że \textit{embedding distillation} wyraźnie pomaga słabszym modelom, nie potwierdziła się.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Wnioski}
\begin{itemize}
    \item Embedding distillation nie przyniosła oczekiwanej, wyraźnej poprawy.
    \item Dla ResNet-18 zysk był niewielki, a dla tiny\_cnn nie było poprawy.
    \item Najlepszy wynik uzyskał wariant \textbf{logit+cls}, czyli klasyczna destylacja logitów.
    \item Oznacza to, że w naszym ustawieniu \textbf{logity teacher'a okazały się bardziej użyteczne niż same embeddingi końcowe}.
    \item Ze względu na ograniczenia czasowe nie wdrażaliśmy bardziej zaawansowanej destylacji warstw pośrednich, która mogłaby być skuteczniejsza.
\end{itemize}
\end{frame}
